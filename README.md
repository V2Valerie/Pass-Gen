# Pass-Gen
simple cli password generator written on python

# Used Modules
 string <br>
 sys <br>
 random <br>

# Usage
  flags: <br>
 -h   help <br>
 -u   uppercase <br>
 -l   lowercase <br>
 -d   digits <br>
 -p   punctuation <br>
 example: <br>
 python pass-gen.py -u -l -d -p 12 <br>
